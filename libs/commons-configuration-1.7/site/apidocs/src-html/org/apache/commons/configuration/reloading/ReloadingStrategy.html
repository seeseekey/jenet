018    package org.apache.commons.configuration.reloading;
019    
020    import org.apache.commons.configuration.FileConfiguration;
021    
022    /**
023     * A strategy to decide if a configuration should be reloaded.
024     *
025     * @author Emmanuel Bourg
026     * @author Olivier Heger
027     * @version $Revision: 439648 $, $Date: 2006-09-02 22:42:10 +0200 (Sa, 02. Sep 2006) $
028     * @since 1.1
029     */
030    public interface ReloadingStrategy
031    {
032        /**
033         * Set the configuration managed by this strategy.
034         *
035         * @param configuration the configuration to monitor
036         */
037        void setConfiguration(FileConfiguration configuration);
038    
039        /**
040         * Initialize the strategy.
041         */
042        void init();
043    
044        /**
045         * Tell if the evaluation of the strategy requires to reload the configuration.
046         *
047         * @return a flag whether a reload should be performed
048         */
049        boolean reloadingRequired();
050    
051        /**
052         * Notify the strategy that the file has been reloaded.
053         */
054        void reloadingPerformed();
055    
056    }
