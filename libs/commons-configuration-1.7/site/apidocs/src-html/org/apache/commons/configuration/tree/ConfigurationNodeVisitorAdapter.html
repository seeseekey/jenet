017    package org.apache.commons.configuration.tree;
018    
019    /**
020     * <p>
021     * A simple adapter class that simplyfies writing custom node visitor
022     * implementations.
023     * </p>
024     * <p>
025     * This class provides dummy implementations for the methods defined in the
026     * <code>ConfigurationNodeVisitor</code> interface. Derived classes only need
027     * to override the methods they really need.
028     * </p>
029     *
030     * @author Oliver Heger
031     */
032    public class ConfigurationNodeVisitorAdapter implements
033            ConfigurationNodeVisitor
034    {
035        /**
036         * Empty dummy implementation of this interface method.
037         *
038         * @param node the node
039         */
040        public void visitBeforeChildren(ConfigurationNode node)
041        {
042        }
043    
044        /**
045         * Empty dummy implementation of this interface method.
046         *
047         * @param node the node
048         */
049        public void visitAfterChildren(ConfigurationNode node)
050        {
051        }
052    
053        /**
054         * Dummy implementation of this interface method. Returns always <b>false</b>.
055         *
056         * @return the terminate flag
057         */
058        public boolean terminate()
059        {
060            return false;
061        }
062    }
